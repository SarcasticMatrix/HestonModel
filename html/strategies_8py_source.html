21def time_varying_strategy(
22 premium_volatility_risk: float,
23 p: float,
24 heston: Heston,
25 V: np.array
26 ) -> np.array:
27 """
28 Function to determine the allocation of the portfolio based on premium volatility risk and p.
29
30 Args:
31 - premium_volatility_risk (float): Premium volatility risk parameter.
32 - p (float): Parameter.
33
34 Returns:
35 - np.array: Number of stocks over time to hold.
36 """
37
38 alpha = heston.r + np.sqrt(V)
39 returns = heston.r + premium_volatility_risk * np.sqrt(V)

45def optimal_allocate_strategy(
46 heston:Heston,
47 p:float,
48 time:np.array
49 ) -> np.array:
50 """
51 Implement the optimal allocation of the portfolio based on premium volatility risk and p.
52
53 Args:
54 - premium_volatility_risk (float): Premium volatility risk parameter.
55 - p (float): Parameter.
56
57 Returns:
58 - np.array: Number of stocks over time to hold over time.
59 """
60
61 k0 = p * heston.premium_volatility_risk**2 / (1-p)
62 k1 = heston.kappa - p * heston.premium_volatility_risk * heston.sigma * heston.rho / (1-p)
63 k2 = heston.sigma**2 + p * heston.sigma**2 * heston.rho**2 /(1-p)
64 k3 = np.sqrt(k1**2 - k0*k2)
65
66 b = lambda t: k0 * (np.exp(k3 * (heston.T-t)) - 1) / (np.exp(k3 * (heston.T-t)) * (k1 + k3) - k1 + k3)
67 pi = lambda t: heston.premium_volatility_risk / (1-p) + (heston.sigma * heston.rho) / (1-p) * b(t)
68
69 allocation = pi(time)
70 return allocation
