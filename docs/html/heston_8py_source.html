12class Heston:
13 """
14 Class to represent a Heston Model: can simulate trajectories and price call options with this underlying.
15
16 The Heston model is a popular stochastic volatility model used to describe the evolution of asset prices,
17 particularly in the context of options pricing. This class allows for the simulation of price paths
18 as well as the pricing of European call options using both Monte Carlo simulations and Fourier transform techniques.
19
20 Attributes:
21 - spot (float): Spot price of the underlying asset.
22 - vol_initial (float): Initial variance of the asset's returns.
23 - r (float): Risk-free interest rate.
24 - kappa (float): Speed of mean reversion of the variance.
25 - theta (float): Long-term average variance.
26 - drift_emm (float): Drift term for the equivalent martingale measure.
27 - sigma (float): Volatility of the variance process.
28 - rho (float): Correlation between the Brownian motions driving the asset and variance.
29 - T (float): Time to maturity for the options being priced.
30 - K (float): Strike price of the options.
31 - premium_volatility_risk (float): Premium for volatility risk (default is 0.0).
32 - seed (int): Seed for random number generation (default is 42).
33
34 Methods:
35 - simulate(self, scheme: str = "euler", n: int = 100, N: int = 1000):
36 Simulates and returns several simulated paths following the Heston model.
37 - monte_carlo_price(self, scheme: str = "euler", n: int = 100, N: int = 1000):
38 Simulates sample paths and estimates the call price with a simple Monte Carlo Method.
39 - fourier_transform_price(self, t=0):
40 Computes the price of a European call option on the underlying asset S following a Heston model using the Heston formula.
41 - carr_madan_price(self):
42 Computes the price of a European call option on the underlying asset S following a Heston model using Carr-Madan Fourier pricing.
43 - plot_simulation(self, scheme: str = 'euler', n: int = 1000):
44 Plots the simulation of a Heston model trajectory.
45 - price_surface(self):
46 Plot the Price of call option as a function of strike and time to maturity.

147 def simulate(self,
148 scheme : str = "euler",
149 n: int = 100,
150 N:int = 1000,
151 ) -> tuple:
152 """
153 Simulates multiple paths according to the Heston model.
154
155 Parameters:
156 - scheme (str): The discretization method to be used (e.g., "euler" or "milstein").
157 - n (int): The number of discrete points in each path.
158 - N (int): The total number of simulated paths.
159
160 Returns:
161 - S (np.array): Array of simulated stock price paths.
162 - V (np.array): Array of simulated variance paths.
163 - null_variance (int): Count of instances where simulated variance equals zero.
164 """
165 random.seed(self.seed)
166
167 dt = self.T / n
168 S = np.zeros((N, n + 1))
169 V = np.zeros((N, n + 1))
170 S[:, 0] = self.spot
171 V[:, 0] = self.vol_initial
172
173 null_variance = 0
174
175 for i in range(1, n+1):
176
177 # Apply reflection scheme
178 if np.any(V[:, i-1] < 0):
179 V[:, i-1] = np.abs(V[:, i-1])
180
181 if np.any(V[:, i-1] == 0):
182 null_variance += np.sum(V[i-1, :] == 0)
183
184 # Brownian motion
185 N1 = np.random.normal(loc=0, scale=1, size=N)
186 N2 = np.random.normal(loc=0, scale=1, size=N)
187 ZV = N1 * np.sqrt(dt)
188 ZS = (self.rho * N1 + np.sqrt(1-self.rho**2) * N2) * np.sqrt(dt)
189
190 # Update the processes
191 #S[:, i] = S[:, i-1] + self.r * S[:, i-1] * dt + np.sqrt(V[:, i-1]) * S[:, i-1] * ZS
192 S[:, i] = S[:, i-1] + (self.r + self.premium_volatility_risk * np.sqrt(V[:, i-1]))* S[:, i-1] * dt + np.sqrt(V[:, i-1]) * S[:, i-1] * ZS
193
194 V[:, i] = V[:, i-1] + (self.kappa * (self.theta - V[:, i-1]) - self.drift_emm * V[:, i-1]) * dt + self.sigma * np.sqrt(V[:, i-1]) * ZV
195 if scheme == "milstein":
196 S[:, i] += 1/2 * V[:, i-1] * S[:, i-1] * (ZS**2 - dt)
197 #S[:, i] += 1/4 * S[:, i-1]**2 * (ZS**2 - dt)
198 V[:, i] += 1/4 * self.sigma**2 * (ZV**2 - dt)
199 elif scheme == 'euler':
200 pass
201 else:
202 print("Choose a scheme between: 'euler' or 'milstein'")
203
204 if N == 1:
205 S = S.flatten()
206 V = V.flatten()
207
208 return S, V, null_variance

210 def monte_carlo_price(self,
211 scheme: str = "euler",
212 n: int = 100,
213 N: int = 1000
214 ) -> float:
215 """
216 Simulates paths to estimate the price of a European call option using the Monte Carlo method.
217
218 This method calculates the option price by averaging the discounted payoffs from simulated asset price paths.
219
220 Parameters:
221 - scheme (str): The discretization method used ("euler" or "milstein").
222 - n (int): Number of discrete points in a path.
223 - N (int): Number of paths to simulate.
224
225 Returns:
226 - result (namedtuple): Contains the following:
227 - price (float): Estimated price of the call option.
228 - standard_deviation (float): Standard deviation of the option payoffs.
229 - infimum (float): Lower bound of the confidence interval.
230 - supremum (float): Upper bound of the confidence interval.
231 """
232 random.seed(self.seed)
233
234 S, _, null_variance = self.simulate(scheme, n, N)
235 print(f"Variance has been null {null_variance} times over the {n*N} iterations ({round(null_variance/(n*N)*100,2)}%) ")
236
237 ST = S[:, -1]
238 payoff = np.maximum(ST - self.K, 0)
239 discounted_payoff = np.exp(-self.r * self.T) * payoff
240
241 price = np.mean(discounted_payoff)
242 standard_deviation = np.std(discounted_payoff, ddof=1)/np.sqrt(N)
243 infimum = price - 1.96 * np.sqrt(standard_deviation / N)
244 supremum = price + 1.96 * np.sqrt(standard_deviation / N)
245
246 Result = namedtuple('Results','price std infinum supremum')
247 return Result(price, standard_deviation, infimum, supremum) # price, standard_deviation, infimum, supremum

249 def characteristic(
250 self,
251 j: int
252 ) -> float:
253 """
254 Creates the characteristic function Psi_j(x, v, t; u) for a given (x, v, t).
255
256 This function returns the characteristic function based on the index provided.
257
258 Parameters:
259 - j (int): Index of the characteristic function (must be 1 or 2).
260
261 Returns:
262 - callable: The characteristic function.
263 """
264
265 if j == 1 :
266 uj = 1/2
267 bj = self.kappa + self.drift_emm - self.rho * self.sigma
268 elif j == 2:
269 uj = - 1/2
270 bj = self.kappa + self.drift_emm
271 else:
272 print('Argument j (int) must be 1 or 2')
273 return 0
274 a = self.kappa * self.theta
275
276 dj = lambda u : np.sqrt((self.rho * self.sigma * u * 1j - bj)**2 - self.sigma**2 * (2 * uj * u * 1j - u**2))
277 gj = lambda u : (self.rho * self.sigma * u *1j - bj - dj(u))/(self.rho * self.sigma * u *1j - bj + dj(u))
278
279 Cj = lambda tau, u : self.r * u * tau * 1j + a/self.sigma**2 * ((bj - self.rho * self.sigma * u * 1j + dj(u)) * tau - 2 * np.log((1-gj(u)*np.exp(dj(u)*tau))/(1-gj(u))))
280 Dj = lambda tau, u : (bj - self.rho * self.sigma * u * 1j + dj(u))/self.sigma**2 * (1-np.exp(dj(u) * tau))/(1-gj(u) * np.exp(dj(u)*tau))
281
282 return lambda x, v, t, u : np.exp(Cj(self.T-t,u) + Dj(self.T-t,u)*v + u * x * 1j)

284 def fourier_transform_price(
285 self,
286 t = 0
287 ):
288 """
289 Calculates the price of a European call option using the Heston formula.
290
291 This method computes the option price by evaluating the integral of the characteristic function.
292
293 Parameters:
294 - t (float): The current time for pricing (default is 0).
295
296 Returns:
297 - price (float): The calculated option price.
298 - error (float): The error associated with the option price calculation.
299 """
300
301 x = np.log(self.spot)
302 v = self.vol_initial
303
304 psi1 = self.characteristic(j=1)
305 integrand1 = lambda u : np.real((np.exp(-u * np.log(self.K) * 1j) * psi1(x, v, t, u))/(u*1j))
306 Q1 = 1/2 + 1/np.pi * quad(func = integrand1, a = 0, b = 1000)[0]
307 error1 = 1/np.pi * quad(func = integrand1, a = 0, b = 1000)[1]
308
309 psi2 = self.characteristic(j=2)
310 integrand2 = lambda u : np.real((np.exp(-u * np.log(self.K) * 1j) * psi2(x, v, t, u))/(u*1j))
311 Q2 = 1/2 + 1/np.pi * quad(func = integrand2, a = 0, b = 1000)[0]
312 error2 = 1/np.pi * quad(func = integrand2, a = 0, b = 1000)[1]
313
314 price = self.spot * Q1 - self.K * np.exp(-self.r * (self.T - t)) * Q2
315 error = self.spot * error1 + self.K * np.exp(-self.r * (self.T - t)) * error2
316 return price, error

318 def carr_madan_price(self):
319 """
320 Computes the price of a European call option using the Carr-Madan Fourier pricing method.
321
322 This method employs the Carr-Madan approach, leveraging the characteristic function to calculate
323 the option price.
324
325 Returns:
326 - price (float): The calculated option price.
327 - error (float): The error associated with the option price calculation.
328 """
329
330 x = np.log(self.spot)
331 v = self.vol_initial
332 t = self.T - 1
333 alpha = 0.3
334
335 price_hat = lambda u: np.exp(- self.r * self.T) / (alpha**2 + alpha - u**2 + u * (2 * alpha + 1) * 1j) \
336 * self.characteristic(j=2)(x, v, t, u - (alpha + 1) * 1j)
337
338 integrand = lambda u: np.exp(- u * np.log(self.K) * 1j) * price_hat(u)
339
340 price = np.exp(- alpha * np.log(self.K)) / np.pi * quad(func = integrand, a = 0, b = 50)[0]
341 error = np.exp(- alpha * np.log(self.K)) / np.pi * quad(func = integrand, a = 0, b = 50)[1]
342
343 return price, error
