121 def simulate(
122 self,
123 T: float = 1,
124 scheme: str = "euler",
125 n: int = 100,
126 N: int = 1000,
127 ) -> np.array:
128 """
129 Simulates and returns several simulated paths following the Black-Scholes model.
130 Parameters:
131 - T (float): Time to maturity in years
132 - scheme (str): the discretization scheme used ('euler' or 'milstein')
133 - n (int): number of time steps in a path
134 - N (int): number of simulated paths
135 Returns:
136 - S (np.array): simulated stock paths
137 """
138 np.random.seed(self.seed)
139
140 dt = T / n
141 S = np.zeros((N, n + 1))
142 S[:, 0] = self.spot
143
144 for i in range(1, n + 1):
145
146 # Brownian motion
147 N1 = np.random.normal(loc=0, scale=1, size=N)
148 Z = N1 * np.sqrt(dt)
149
150 # Update the processes
151 S[:, i] = S[:, i - 1] + self.mu * S[:, i - 1] * dt + self.volatility * S[:, i - 1] * Z
152
153 if scheme == "milstein":
154 S[:, i] += 1 / 2 * self.volatility * S[:, i - 1] * (Z ** 2 - dt)
155 elif scheme != 'euler':
156 print("Choose a scheme between: 'euler' or 'milstein'")
157
158 return S

188 def call_price(
189 self,
190 strike: float,
191 spot: float = None,
192 r: float = None,
193 volatility: float = None,
194 T: float = 1,
195 ):
196
197 """
198 Calculates the price of a European call option using the Black-Scholes formula.
199 Parameters:
200 - spot (float): spot value of the underlying asset
201 - r (float): risk-free interest rate
202 - volatility (float): volatility of the underlying asset
203 - T (float): time to expiration (in years)
204 - strike (float): strike price of the option
205 Returns:
206 - call_price (float): price of the European call option
207 """
208 if spot is None:
209 spot = self.spot
210 if r is None:
211 r = self.r
212 if volatility is None:
213 volatility = self.volatility
214
215
216 d1 = (np.log(spot / strike) + (r + 0.5 * volatility ** 2) * T) / (volatility * np.sqrt(T))
217 d2 = d1 - volatility * np.sqrt(T)
218 call_price = spot * norm.cdf(d1) - strike * np.exp(-r * T) * norm.cdf(d2)
219 return call_price

378 def delta_hedging(
379 self,
380 flag_option: Literal['call', 'put'],
381 strike: float,
382 hedging_volatility: float,
383 n: float = 1000,
384 N: float = 100
385 ):
386 """
387 Implement a delta hedging strategy using both a risky asset (underlying asset)
388 and a non-risky asset for a European option.
389 Parameters:
390 - flag_option (str):
391 Type of option. Should be 'call' for a call option or 'put' for a put option.
392 - strike (float):
393 The strike price of the option.
394 - hedging_volatility (float):
395 The volatility used for hedging purposes.
396 - n (float, optional):
397 The number of simulation steps or trading intervals over the life
398 of the option. Defaults to 1000. This parameter controls how often
399 the portfolio is rebalanced to maintain a delta-neutral position.
400 - N (float, optional):
401 The number of simulations.
402 Returns:
403 - portfolio (np.array): allocation,
404 - S (np.array):
405 """
406
407 time = np.linspace(start=0, stop=T, num=n+1)
408 dt = T / n
409
410 S = self.simulate(scheme='milstein', n=n, N=N)
411 portfolio = np.zeros_like(S)
412
413 if flag_option == 'call':
414 portfolio[:,0] = self.call_price(strike=strike, spot=S[:,0], volatility=hedging_volatility)
415 else:
416 portfolio[:,0] = self.put_price(strike=strike, spot=S[:,0], volatility=hedging_volatility)
417
418 stocks = self.delta(
419 spot=S[:,0],
420 T=T,
421 volatility=hedging_volatility,
422 strike=strike,
423 flag_option=flag_option,
424 )
425 bank = portfolio[:, 0] - stocks * S[:, 0]
426
427 for t in range(1, n):
428
429 portfolio[:, t] = stocks * S[:, t] + bank * np.exp(dt * self.r)
430 stocks = self.delta(
431 spot = S[:, t],
432 T = T - time[t],
433 volatility = hedging_volatility,
434 strike = strike,
435 flag_option = flag_option,
436 )
437 bank = portfolio[:, t] - stocks * S[:, t]
438
439 portfolio[:, -1] = stocks * S[:,-1] + bank * np.exp(dt * self.r)
440
441 return portfolio, S
