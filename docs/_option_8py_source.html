99 def price_put(self, flag_model: Literal['heston', 'blackScholes'], params: list) -> float:
100 """
101 Calculates the price of a put option using the specified model.
102
103 Parameters:
104 flag_model (Literal['heston', 'blackScholes']): The model to use for pricing ('heston' or 'blackScholes').
105 params (list): The parameters required by the chosen model. For 'heston', this should include
106 [kappa, theta, drift_emm, sigma, rho]. For 'blackScholes', this should be [mu].
107
108 Returns:
109 float: The price of the put option.
110 """
111 call_price = self.price_call(flag_model, params)
112 price = call_price - self.spot + self.strike * np.exp(- self.interest * self.time_to_maturity)
113 return price
