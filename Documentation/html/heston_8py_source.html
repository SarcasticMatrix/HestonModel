56 def simulate(self,
57 scheme : str = "euler",
58 n: int = 100,
59 N:int = 1000,
60 ) -> tuple:
61 # generateHestonPathEulerDisc and generateHestonPathMilsteinDisc
62 """
63 Simulates and returns several simulated paths following the Heston model.
64
65 Parameters:
66 - scheme (str): the discretization scheme used
67 - n (int): number of points in a path
68 - N (int): number of simulated paths
69
70 Returns:
71 - S (np.array): stock paths
72 - V (np.array): variance paths
73 - null_variance (int): number of times the simulated variance has been null
74 """
75 random.seed(self.seed)
76
77 dt = self.T / n
78 S = np.zeros((N, n + 1))
79 V = np.zeros((N, n + 1))
80 S[:, 0] = self.S0
81 V[:, 0] = self.V0
82
83 null_variance = 0
84
85 for i in range(1, n+1):
86
87 # Apply reflection scheme
88 if np.any(V[:, i-1] < 0):
89 V[:, i-1] = np.abs(V[:, i-1])
90
91 if np.any(V[:, i-1] == 0):
92 null_variance += np.sum(V[i-1, :] == 0)
93
94 # Brownian motion
95 N1 = np.random.normal(loc=0, scale=1, size=N)
96 N2 = np.random.normal(loc=0, scale=1, size=N)
97 ZV = N1 * np.sqrt(dt)
98 ZS = (self.rho * N1 + np.sqrt(1-self.rho**2) * N2) * np.sqrt(dt)
99
100 # Update the processes
101 #S[:, i] = S[:, i-1] + self.r * S[:, i-1] * dt + np.sqrt(V[:, i-1]) * S[:, i-1] * ZS
102 S[:, i] = S[:, i-1] + (self.r + self.premium_volatility_risk * np.sqrt(V[:, i-1]))* S[:, i-1] * dt + np.sqrt(V[:, i-1]) * S[:, i-1] * ZS
103
104 V[:, i] = V[:, i-1] + (self.kappa * (self.theta - V[:, i-1]) - self.drift_emm * V[:, i-1]) * dt + self.sigma * np.sqrt(V[:, i-1]) * ZV
105 if scheme == "milstein":
106 S[:, i] += 1/2 * V[:, i-1] * S[:, i-1] * (ZS**2 - dt)
107 #S[:, i] += 1/4 * S[:, i-1]**2 * (ZS**2 - dt)
108 V[:, i] += 1/4 * self.sigma**2 * (ZV**2 - dt)
109 elif scheme == 'euler':
110 pass
111 else:
112 print("Choose a scheme between: 'euler' or 'milstein'")
113
114 if N == 1:
115 S = S.flatten()
116 V = V.flatten()
117
118 return S, V, null_variance

121 scheme: str = "euler",
122 n: int = 100,
123 N: int = 1000
124 ) -> float:
125 # priceHestonCallViaEulerMC and priceHestonCallViaMilsteinMC
126 """
127 Simulates sample paths and estimates the call price with a simple Monte Carlo Method.
128
129 Parameters:
130 - scheme (str): the discretization scheme used ("euler" or "milstein")
131 - n (int): number of points in a path
132 - N (int): number of simulated paths
133
134 Returns:
135 - result (namedtuple): with the following attributes:
136 - price (float): estimation by Monte Carlo of the call price
137 - standard_deviation (float): standard deviation of the option payoff
138 - infimum (float): infimum of the confidence interval
139 - supremum (float): supremum of the confidence interval
140 """
141 random.seed(self.seed)
142
143 S, _, null_variance = self.simulate(scheme, n, N)
144 print(f"Variance has been null {null_variance} times over the {n*N} iterations ({round(null_variance/(n*N)*100,2)}%) ")
145
146 ST = S[:, -1]
147 payoff = np.maximum(ST - self.K, 0)
148 discounted_payoff = np.exp(-self.r * self.T) * payoff
149
150 price = np.mean(discounted_payoff)
151 standard_deviation = np.std(discounted_payoff, ddof=1)/np.sqrt(N)
152 infimum = price - 1.96 * np.sqrt(standard_deviation / N)
153 supremum = price + 1.96 * np.sqrt(standard_deviation / N)
154
155 Result = namedtuple('Results','price std infinum supremum')
156 return Result(price, standard_deviation, infimum, supremum) # price, standard_deviation, infimum, supremum

158 def characteristic(
159 self,
160 j: int
161 ) -> float:
162 """
163 Create the characteristic function Psi_j(x, v, t; u), for a given (x, v, t).
164
165 Parameters:
166 - j (int): index of the characteristic function
167
168 Returns:
169 - callable: characteristic function
170 """
171
172 if j == 1 :
173 uj = 1/2
174 bj = self.kappa + self.drift_emm - self.rho * self.sigma
175 elif j == 2:
176 uj = - 1/2
177 bj = self.kappa + self.drift_emm
178 else:
179 print('Argument j (int) must be 1 or 2')
180 return 0
181 a = self.kappa * self.theta
182
183 dj = lambda u : np.sqrt((self.rho * self.sigma * u * 1j - bj)**2 - self.sigma**2 * (2 * uj * u * 1j - u**2))
184 gj = lambda u : (self.rho * self.sigma * u *1j - bj - dj(u))/(self.rho * self.sigma * u *1j - bj + dj(u))
185
186 Cj = lambda tau, u : self.r * u * tau * 1j + a/self.sigma**2 * ((bj - self.rho * self.sigma * u * 1j + dj(u)) * tau - 2 * np.log((1-gj(u)*np.exp(dj(u)*tau))/(1-gj(u))))
187 Dj = lambda tau, u : (bj - self.rho * self.sigma * u * 1j + dj(u))/self.sigma**2 * (1-np.exp(dj(u) * tau))/(1-gj(u) * np.exp(dj(u)*tau))
188
189 return lambda x, v, t, u : np.exp(Cj(self.T-t,u) + Dj(self.T-t,u)*v + u * x * 1j)

191 def fourier_transform_price(
192 self,
193 t = 0
194 ):
195 """
196 Computes the price of a European call option on the underlying asset S following a Heston model using the Heston formula.
197
198 Parameters:
199 - t (float): time
200
201 Returns:
202 - price (float): option price
203 - error (float): error in the option price computation
204 """
205
206 x = np.log(self.S0)
207 v = self.V0
208
209 psi1 = self.characteristic(j=1)
210 integrand1 = lambda u : np.real((np.exp(-u * np.log(self.K) * 1j) * psi1(x, v, t, u))/(u*1j))
211 Q1 = 1/2 + 1/np.pi * quad(func = integrand1, a = 0, b = 1000)[0]
212 error1 = 1/np.pi * quad(func = integrand1, a = 0, b = 1000)[1]
213
214 psi2 = self.characteristic(j=2)
215 integrand2 = lambda u : np.real((np.exp(-u * np.log(self.K) * 1j) * psi2(x, v, t, u))/(u*1j))
216 Q2 = 1/2 + 1/np.pi * quad(func = integrand2, a = 0, b = 1000)[0]
217 error2 = 1/np.pi * quad(func = integrand2, a = 0, b = 1000)[1]
218
219 price = self.S0 * Q1 - self.K * np.exp(-self.r * (self.T - t)) * Q2
220 error = self.S0 * error1 + self.K * np.exp(-self.r * (self.T - t)) * error2
221 return price, error

223 def carr_madan_price(self):
224 """
225 Computes the price of a European call option on the underlying asset S following a Heston model using Carr-Madan Fourier pricing.
226
227 Returns:
228 - price (float): option price
229 - error (float): error in the option price computation
230 """
231
232 x = np.log(self.S0)
233 v = self.V0
234 t = self.T - 1
235 alpha = 0.3
236
237 price_hat = lambda u: np.exp(- self.r * self.T) / (alpha**2 + alpha - u**2 + u * (2 * alpha + 1) * 1j) \
238 * self.characteristic(j=2)(x, v, t, u - (alpha + 1) * 1j)
239
240 integrand = lambda u: np.exp(- u * np.log(self.K) * 1j) * price_hat(u)
241
242 price = np.exp(- alpha * np.log(self.K)) / np.pi * quad(func = integrand, a = 0, b = 50)[0]
243 error = np.exp(- alpha * np.log(self.K)) / np.pi * quad(func = integrand, a = 0, b = 50)[1]
244
245 return price, error

316 history = stock.history(period="1d")

326 option_chain = stock.option_chain(exp_date)

337 maturities.append([maturity] * liquidity)

339 volumes = np.hstack(volumes)
340 strikes = np.hstack(strikes)

346 is_nan_prices = np.isnan(prices)

348 mask = ~ mask

352 heston.theta = x[1]

354 heston.rho = x[3]
355 heston.drift_emm = x[4]
356 heston.V0 = x[5]

365 model_prices = np.array(model_prices)
366 result = np.sum(
367 volumes[mask] * np.abs(prices[mask] - model_prices[mask])
368 )
369
370 return result

372 print('Callibration is running...')
373 res = minimize(fun=objective_function, x0=x0)
374
